LeaveService.java — @Transactional 트랜잭션 무결성 Java · Spring Boot 
// ****** 운영중인 시스템의 실제 코드가 아닌, 일부 기능을 재현한 코드입니다. ******

@Service
@Slf4j
public class LeaveService {

    private final LeaveMapper    leaveMapper;
    private final LeaveHistMapper histMapper;
    private final AlarmMapper     alarmMapper;

    /**
     * 휴가 신청 — 3개 테이블 업데이트를 하나의 트랜잭션으로 묶어 원자성 보장.
     * 예외 발생 시 전체 롤백되므로 부분 저장 현상 없음.
     */
    @Transactional
    public void applyLeave(LeaveApplyDto dto) {

        // 1. 잔여 휴가 차감 (T_LEAVE_BALANCE)
        int remaining = leaveMapper.selectRemainingDays(dto.getEmpNo());
        if (remaining < dto.getDays()) {
            throw new BusinessException("잔여 휴가 일수가 부족합니다.");
        }
        leaveMapper.deductLeaveBalance(dto);

        // 2. 휴가 신청 등록 (T_LEAVE_REQUEST)
        leaveMapper.insertLeaveRequest(dto);

        // 3. 신청 이력 적재 (T_LEAVE_HISTORY)
        histMapper.insertHistory(LeaveHistDto.of(dto, "APPLY"));

        // 4. 결재자 알림 등록 (T_ALARM)
        // → 예외 시 여기까지 포함해 전체 롤백
        alarmMapper.insertApprovalAlarm(dto.getApproverEmpNo(), dto);

        log.info("[LEAVE] 휴가 신청 완료 — empNo: {}, days: {}", dto.getEmpNo(), dto.getDays());
    }

    /**
     * 휴가 승인 — 승인 상태 변경 + 이력 적재 원자성 보장.
     * propagation 기본값(REQUIRED): 상위 트랜잭션 있으면 합류, 없으면 새로 생성.
     */
    @Transactional
    public void approveLeave(LeaveApproveDto dto) {
        leaveMapper.updateLeaveStatus(dto.getLeaveId(), "APPROVED");
        histMapper.insertHistory(LeaveHistDto.of(dto, "APPROVE"));
        alarmMapper.insertAlarm(dto.getApplicantEmpNo(), "휴가가 승인되었습니다.");
    }
}
핵심 포인트: 잔여 휴가 차감 → 신청 등록 → 이력 적재 → 알림 등록의 4단계 중 어느 한 곳에서 예외가 발생해도 @Transactional이 전체를 롤백하여 데이터 정합성을 보장합니다. 기존에는 트랜잭션 없이 개별 쿼리가 실행되어 부분 저장 문제가 발생했습니다.
treeGridSearch.js — 공통 트리 그리드 검색 모듈 JavaScript · ES6+ 
// ****** 운영중인 시스템의 실제 코드가 아닌, 일부 기능을 재현한 코드입니다. ******

// ── treeGridSearch.js — 여러 시스템에서 공통으로 사용하는 트리 그리드 검색 모듈 ─
/**
 * TreeGridSearch: tuiGrid 트리 그리드의 검색·확장·축소를 공통 처리하는 모듈.
 * 각 시스템 화면에서 gridInstance와 searchKey만 넘기면 동작.
 *
 * 사용법:
 *   import { TreeGridSearch } from '/common/js/treeGridSearch.js';
 *   const searcher = new TreeGridSearch(gridInstance, { searchKey: 'deptNm' });
 *   searcher.search('개발');
 */
export class TreeGridSearch {

    constructor(grid, options = {}) {
        this.grid      = grid;
        this.searchKey = options.searchKey || 'name';
        this.matchedRowKeys = [];
    }

    /** 검색어로 매칭된 행 하이라이트 + 상위 노드 자동 확장 */
    search(keyword) {
        this.reset();
        if (!keyword || !keyword.trim()) return;

        const allRows = this.grid.getData();
        const keyword_lower = keyword.toLowerCase();

        allRows.forEach(row => {
            const val = String(row[this.searchKey] || '').toLowerCase();
            if (val.includes(keyword_lower)) {
                this.matchedRowKeys.push(row.rowKey);
                // 매칭 행의 모든 상위 노드를 재귀적으로 확장
                this._expandAncestors(row.rowKey);
                this.grid.addCellClassName(row.rowKey, this.searchKey, 'search-highlight');
            }
        });
    }

    /** 상위 노드 재귀 확장 */
    _expandAncestors(rowKey) {
        const row = this.grid.getRow(rowKey);
        if (row && row._treeData && row._treeData.parentRowKey != null) {
            this.grid.expandTree(row._treeData.parentRowKey, false);
            this._expandAncestors(row._treeData.parentRowKey); // 재귀
        }
    }

    /** 검색 초기화 */
    reset() {
        this.matchedRowKeys.forEach(key => {
            this.grid.removeCellClassName(key, this.searchKey, 'search-highlight');
        });
        this.matchedRowKeys = [];
        this.grid.collapseAll();
    }
}

// ── 화면에서 사용 예시 (dept-tree.html) ───────────────────────────────
import { TreeGridSearch } from '/common/js/treeGridSearch.js';

const searcher = new TreeGridSearch(deptGrid, { searchKey: 'deptNm' });

document.getElementById('searchInput').addEventListener('input', e => {
    searcher.search(e.target.value);
});
모듈화 효과: 기존에는 시스템별로 동일한 트리 그리드 검색 로직이 중복 구현되어 있었습니다. 공통 JS 모듈로 추출·배포 후 각 화면에서는 gridInstance와 searchKey만 전달하면 되어 유지보수 비용을 크게 절감했습니다.

/**
 * 휴가 신청 처리:
 * 1) 잔여 일수 검증 (API 완료 대기)
 * 2) 검증 통과 후 신청 API 호출
 * 3) 결과에 따라 다국어 메시지 노출
 *
 * async/await으로 선행 프로세스 완료 후 후속 실행 — 경쟁 상태 방지
 */
async function submitLeave() {
    const dto = {
        leaveType : document.getElementById('leaveType').value,
        startDate : document.getElementById('startDate').value,
        endDate   : document.getElementById('endDate').value,
        days      : calcDays(),
        reason    : document.getElementById('reason').value,
    };

    try {
        showLoading(true);

        // Step 1: 잔여 일수 검증 — 완료 후 Step 2 진행
        const checkResult = await $.ajax({
            url  : '/api/leave/check-balance',
            type : 'POST',
            contentType: 'application/json',
            data : JSON.stringify({ days: dto.days })
        });

        if (!checkResult.sufficient) {
            alert(msg('leave.apply.error.balance'));  // i18n 메시지
            return;
        }

        // Step 2: 신청 API — Step 1 완료 보장 후 실행
        await $.ajax({
            url  : '/api/leave/apply',
            type : 'POST',
            contentType: 'application/json',
            data : JSON.stringify(dto)
        });

        alert(msg('leave.apply.success'));  // i18n 메시지
        location.reload();

    } catch (e) {
        console.error('Leave apply error:', e);
        alert('처리 중 오류가 발생했습니다.');
    } finally {
        showLoading(false);
    }
}

// i18n 메시지 헬퍼 — Thymeleaf가 렌더링한 messages 객체에서 조회
function msg(key) { return window.MESSAGES[key] || key; }
다국어 + 비동기: Spring MessageSource + Thymeleaf #{...} 표현식으로 서버 렌더링 시 다국어 처리. 클라이언트 동적 메시지는 window.MESSAGES 객체로 전달해 JS에서도 i18n 적용. async/await로 잔여일수 검증 API 완료 후 신청 API를 실행하여 순서 보장 및 경쟁 상태를 방지합니다.
