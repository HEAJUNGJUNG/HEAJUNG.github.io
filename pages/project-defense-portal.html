방산 협력사의 보안 자산 관리 시스템을 풀스택으로 신규 개발하고 운영 배포까지 담당한 프로젝트입니다. Spring Batch 기반 SMS·이메일 자동 알림, POST API 보안 설계, CSV 대량 업로드 파이프라인을 단독으로 설계·구현하였으며 모의해킹 대응 및 Linux 서버 배포 운영을 병행했습니다.

// ── AssetBatchService.java  반납일 지난 자산유무 확인 및 알림메일 발송 서비스 로직 ───────────────────────────────────────────
@Service
@Slf4j
public class AssetBatchService {

    private final AssetMapper assetMapper;
    private final MailService mailService;

    public AssetBatchService(AssetMapper assetMapper, MailService mailService) {
        this.assetMapper = assetMapper;
        this.mailService = mailService;
    }

    public Map checkOverdueAssetsAndSendMail() {
        Map summary = new HashMap<>();
        
        // 1. 반납일 지난 예약건 조회
        List> userList = assetMapper.selectOverdueAssetUsers();
        if (userList == null || userList.isEmpty()) {
            summary.put("totalCount", 0);
            summary.put("successCount", 0);
            summary.put("message", "조회된 데이터가 없습니다.");
            return summary;
        }

        // 2. 메일 템플릿 조회
        String template = assetMapper.selectMailTemplate("RETURN_OVERDUE");
        if (template == null || template.isEmpty()) {
            log.error("[BATCH ERROR] 메일 템플릿 누락");
            throw new RuntimeException("Mail template (RETURN_OVERDUE) is missing in DB");
        }

        int successCount = 0;

        // 3. 루프 처리 및 메일 발송
        for (Map user : userList) {
            String email = (String) user.get("EMAIL");
            
            if (email == null || email.isBlank()) {
                log.error("[BATCH SKIP] 이메일 정보 없음: {}", user.get("USER_NAME"));
                continue;
            }

            try {
                Map mailParams = new HashMap<>(user);
                mailParams.put("content", template);
                
                mailService.sendMail(mailParams);
                successCount++;
            } catch (Exception e) {
                log.error("[BATCH FAIL] 메일 발송 오류 ({}): {}", user.get("USER_NAME"), e.getMessage());
            }
        }

        summary.put("totalCount", userList.size());
        summary.put("successCount", successCount);
        summary.put("status", "COMPLETED");

        log.info("[BATCH RESULT] Total: {}, Success: {}", userList.size(), successCount);
        return summary;
    }
}
// ── BatchController.java  배치 코드로 서비스를 구분 , 할당된 배치 코드로 해당하는 서비스를 호출 ────────────────────────────────
@RestController
@RequestMapping("/api/batch")
@Slf4j
public class BatchController {

    private final AssetBatchService assetBatchService;

    public BatchController(AssetBatchService assetBatchService) {
        this.assetBatchService = assetBatchService;
    }

    @PostMapping("/execute")
    public ResponseEntity> executeBatchByCode(@RequestParam String batchCode) {
        log.info("Batch Execution Triggered: {}", batchCode);
        
        Map response = new HashMap<>();

        if ("ASSET_01".equals(batchCode)) {
            try {
                // 서비스 호출 및 결과 데이터 수신
                Map result = assetBatchService.checkOverdueAssetsAndSendMail();
                response.put("status", "SUCCESS");
                response.put("data", result);
                return ResponseEntity.ok(response); // 200 OK
            } catch (Exception e) {
                log.error("Batch execution failed: {}", e.getMessage());
                response.put("status", "ERROR");
                response.put("message", e.getMessage());
                return ResponseEntity.status(HttpStatus.INTERNAL_SERVER_ERROR).body(response); // 500 Error
            }
        } else {
            response.put("status", "FAIL");
            response.put("message", "Invalid Batch Code");
            return ResponseEntity.badRequest().body(response); // 400 Bad Request
        }
    }
}
// ── SchedulerConfig.java  할당된 배치 코드로 컨트롤러에 해당 작업을 언제 실행할건지 cron으로 설정 ──────────────────────────────
@Configuration
@EnableScheduling
public class SchedulerConfig {

    private final BatchController batchController;

    public SchedulerConfig(BatchController batchController) {
        this.batchController = batchController;
    }

    // 매일 오전 6시 30분 실행 (초 분 시 일 월 요일)
    @Scheduled(cron = "0 30 6 * * *")
    public void runAssetReturnBatch() {
        // 공통코드 ASSET_01을 파라미터로 컨트롤러 호출
        batchController.executeBatchByCode("ASSET_01");
    }
}

	 * 대량 데이터 일괄 등록 (100건 단위 Chunk 처리 및 예외 전파)
	 * @param successRows 검증이 완료된 자산 DTO 리스트

	 * @throws Exception 예외 발생 시 catch하지 않고 상위로 던져 트랜잭션 롤백 유도
	 */
    @Transactional
	public ResponseEntity> uploadAssets(List successRows) throws Exception {
		
		// 데이터 부재 시 조기 리턴 (Fast-fail)
		if (successRows == null || successRows.isEmpty()) {
			return ResponseEntity.ok(Map.of("count", 0, "status", "EMPTY")); 
		}

		int totalSize = successRows.size();
		int chunkSize = 100; // 100건 단위로 분할하여 쿼리 길이 및 메모리 과부하 방어

		IntStream.range(0, (totalSize + chunkSize - 1) / chunkSize)
				 .mapToObj(i -> successRows.subList(i * chunkSize, Math.min((i + 1) * chunkSize, totalSize)))
				 .forEach(chunk -> {
					 // MyBatis foreach 일괄 등록 호출

		log.info("[BATCH SUCCESS] 총 {}건의 일괄 등록이 완료되었습니다.", totalSize);

// ── assetUpload.html 화면에서 csv파일을 업로드 후 콜백이 오면 결과팝업 호출 ────────────────────────────────────────────────────		 
/**
 * 자산 일괄 등록 프로세스
 */
async function handleCSVUpload() {
    const fileInput = document.getElementById('csvFile');
    const file = fileInput.files[0];

    // 1. 파일 및 확장자 검사
    if (!file || !file.name.endsWith('.csv')) {
        alert("CSV 파일만 업로드 가능합니다.");
        return;
    }

    try {
        // UI 프리징 방지를 위한 로딩바 활성화
        showLoading(true); 

        // 2. DRM 파일 업로드 프로세스 (비동기 처리)
        // DRM 컴포넌트가 Promise를 반환한다고 가정
        const uploadResult = await new Promise((resolve, reject) => {
            drmComponent.upload(file, (res) => {
                if(res.status === 'OK') resolve(res);
                else reject(new Error("DRM 인증 실패"));
            });
        });

        const fileId = uploadResult.fileId;

        // 3. AJAX를 이용한 CSV 파싱 요청 (검증 단계)
        const parseResponse = await $.ajax({
            url: '/api/asset/csv/parse',
            type: 'POST',
            contentType: 'application/json',
            data: JSON.stringify({ fileId: fileId })
        });

        const { successRows, errorRows } = parseResponse;

        // 4. 검증 결과에 따른 사용자 컨펌
        if (errorRows && errorRows.length > 0) {
            const confirmMsg = `검증 오류 ${errorRows.length}건이 발견되었습니다. 정상 건만 등록하시겠습니까?`;
            if (!confirm(confirmMsg)) return;
        }

        // 5. 최종 일괄 업로드 요청 (Post 방식)
        const finalResult = await $.ajax({
            url: '/api/asset/csv/upload',
            type: 'POST',
            contentType: 'application/json',
            data: JSON.stringify(successRows)
        });

        alert(`${finalResult.count}건의 자산이 등록되었습니다.`);

    } catch (error) {
        console.error("Batch Upload Error:", error);
        alert("처리 중 오류가 발생했습니다.");
    } finally {
        showLoading(false); // 로딩 종료
    }
}
